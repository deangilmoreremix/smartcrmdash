// Main integration
import { allGoals, goalCategories } from '../data/goalsData';

// Filter goals by category
const getGoalsByCategory = (categoryId: string) => {
  return allGoals.filter(goal => 
    goal.category.toLowerCase() === categoryId.toLowerCase()
  );
};

// Search functionality
const searchGoals = (query: string) => {
  return allGoals.filter(goal =>
    goal.title.toLowerCase().includes(query.toLowerCase()) ||
    goal.description.toLowerCase().includes(query.toLowerCase()) ||
    goal.businessImpact.toLowerCase().includes(query.toLowerCase())
  );
};
